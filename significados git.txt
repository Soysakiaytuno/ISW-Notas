git init;
	Crea un repositorio en git;

git config;
	Te deja introducir tus datos como nombres y correo;

git add;
	Agregas un nuevo archivo al repositorio en git; 

git commit;
	Confirmas el archivo ingresado en el repositorio;

git push;
	Envia el conjunto de cambios a un solo conjunto de cambios;

git pull;
	Descargas una rama de un repositorio remoto;

git branch;
	Te deja crear, listar o eliminar ramas;

git checkout;
	Ademas de confirmar ramas te deja navegar entre ellas;

github;
